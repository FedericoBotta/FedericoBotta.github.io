document.addEventListener('DOMContentLoaded', function() {
  var filterBtns = document.querySelectorAll('.filter-btn');
  var pubItems = document.querySelectorAll('.publication-item');

  filterBtns.forEach(function(btn) {
    btn.addEventListener('click', function() {
      var year = this.getAttribute('data-year');

      // Update active button
      filterBtns.forEach(function(b) { b.classList.remove('active'); b.setAttribute('aria-pressed', 'false'); });
      this.classList.add('active');
      this.setAttribute('aria-pressed', 'true');

      // Show/hide publications
      pubItems.forEach(function(item) {
        if (year === 'all' || item.getAttribute('data-year') === year) {
          item.style.display = '';
        } else {
          item.style.display = 'none';
        }
      });
    });
  });
});
</script>
